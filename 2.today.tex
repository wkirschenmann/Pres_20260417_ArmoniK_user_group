	\includegraphics[width=0.88\textwidth]{mermaid-simplifed-architecture.png}

	\note
	{
		\begin{itemize}
			\item \textbf{Control Plane:} Handles session and task lifecycle, scheduling, and state management
			\item \textbf{Data Plane:} Object storage for task inputs, outputs, and intermediate results
			\item \textbf{Compute Plane:} Stateless agents and workers that execute tasks and report results back
			\item Strict separation of planes limits blast radius and enables independent scaling
		\end{itemize}
	}

	\begin{block}{Definition}
		Paradigm focusing on the decomposition of complex operations into smaller tasks
	\end{block}
	\begin{itemize}
		\item Expression of complex data-driven dependencies
		\item ArmoniK's \textbf{distributed scheduler} responsible for:
		      \begin{itemize}
			      \item Task distribution and load balancing
			      \item Dependency resolution
			      \item Tasks execution
			      \item Data management (overlapping, prefetching, and checkpointing)

\begin{frame}
	\frametitle{ArmoniK Programming Model}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\begin{itemize}
				\item \textbf{Task Graph}: Bipartite DAG whose node sets are tasks and data
				\item \textbf{Task node}: Single-node computation (possibly multi-threaded) taking one or several data inputs and outputting one or several data
				\item \textbf{Data node}: Immutable piece of data depending on only one task at most
				\item \textbf{Dependencies}: Dependencies between tasks are expressed as data dependencies
				\item \textbf{Dynamic}: The graph may not be entirely known in advance (tasks can append tasks and replace edges with subgraphs)
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\centering
			\includegraphics[width=0.95\textwidth]{mermaid-task-graph.png}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{The Client-Worker Pattern}
	\begin{columns}[T]
		\begin{column}{0.5\textwidth}
			\begin{itemize}
				\item \textbf{Client}: User-developed software that submits the initial task graphs to ArmoniK and retrieves results
				\item \textbf{Worker}: User-developed software capable of performing one or several tasks, depending on its implementation:
				      \begin{itemize}
					      \item Can take input data, perform calculations, and return a result
					      \item Can submit new tasks and set the result as output of a new task
				      \end{itemize}
				\item \textbf{Synchronization via Data}: The client might not be aware that the task graph has been modified
			\end{itemize}
		\end{column}
		\begin{column}{0.5\textwidth}
			\centering
			\includegraphics[width=0.95\textwidth]{mermaid-client-worker.png}
		\end{column}
	\end{columns}
\end{frame}

		\item Submissions can happen anytime
		\item Tasks can submit new tasks
		\item Tasks can delegate the production of their output to their new tasks

\begin{frame}{Fault Tolerant}
	\begin{itemize}
		\item \textbf{Automatic Recovery:} Failed tasks are retried or rescheduled transparently
		\item \textbf{Idempotent Operations:} Safe retries with consistent outcomes guaranteed by the task state machine
		\item \textbf{Failure Isolation:} Failures contained to partitions and workers without cascading
		\item \textbf{Graceful Degradation:} ArmoniK services restart without impacting in-flight tasks
	\end{itemize}
\end{frame}

\begin{frame}{ArmoniK v2.23.0 — Performance Improvements}
	\begin{itemize}
		\item \textbf{Filesystem Cache:} Node-local LRU cache stores task inputs and outputs, eliminating redundant Object Storage round-trips when data is reused on the same node; configurable eviction threshold
		\item \textbf{Configurable Streaming Chunks:} Fixed 84\,KB replaced by a server-defined 2\,MB default — fewer gRPC messages for large payloads, lower CPU overhead, dynamically propagated to clients and agents
		\item \textbf{C++ SDK Parallelism:} Upload, download, and wait operations are batched and parallelised via a thread pool, reducing blocking time and network latency
		\item \textbf{.NET\,10 Upgrade:} All orchestrator images move from .NET\,8 to .NET\,10 LTS — improved throughput, reduced memory footprint, no impact on clients/workers running older runtimes
	\end{itemize}
\end{frame}

\begin{frame}{ArmoniK v2.23.0 — Developer Experience \& Ecosystem}
	\begin{itemize}
		\item \textbf{Redesigned C\# SDK:} Blob abstraction for data lifecycle, explicit task-graph definition, and modern async worker model; C\#, C++, and Java SDKs are now officially supported
		\item \textbf{Declarative Initialization:} Partitions, roles, and authentication configured via structured environment variables applied idempotently at startup — no manual database intervention required
		\item \textbf{ArmoniK Load Balancer:} Native Kubernetes deployment with TLS/mTLS, ingress, and GUI integration — officially part of reference deployments
		\item \textbf{NATS JetStream Adapter:} New queue adapter for NATS with durable JetStream persistence
		\item \textbf{Distroless Images:} All orchestrator images switch to distroless base, reducing CVE exposure and image size
	\end{itemize}
\end{frame}
